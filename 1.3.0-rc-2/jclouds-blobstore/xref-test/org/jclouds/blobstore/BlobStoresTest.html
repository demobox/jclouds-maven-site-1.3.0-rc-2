60     @Test
61     public void testListAllFromTransientBlobStore() throws Exception {
62        final int NUM_BLOBS = 31;
63        ListContainerOptions options = ListContainerOptions.Builder.maxResults(10);
64        BlobStoreContext context = new BlobStoreContextFactory().createContext("transient", "dummyid", "dummykey");
65        BlobStore blobStore = null;
66        try {
67           blobStore = context.getBlobStore();
68           blobStore.createContainerInLocation(null, containerName);   
69           Set<String> expectedNames = new HashSet<String>();
70           for (int i = 0; i < NUM_BLOBS; i++) {
71              String blobName = "myname"+i;
72              blobStore.putBlob(containerName, blobStore.blobBuilder(blobName).payload("payload"+i).build());
73              expectedNames.add(blobName);
74           }
75           
76           Iterable<StorageMetadata> iterable = BlobStores.listAll(blobStore, containerName, options);
77           Iterable<String> iterableNames = Iterables.transform(iterable, new Function<StorageMetadata,String>() {
78              @Override public String apply(StorageMetadata input) {
79                 return input.getName();
80              }});
81           
82           // Note that blob.getMetadata being put does not equal blob metadata being retrieved 
83           // because uri is null in one and populated in the other.
84           // Therefore we just compare names to ensure the iterator worked.
85           assertEquals(ImmutableSet.copyOf(iterableNames), expectedNames);
86        } finally {
87           if (blobStore != null) blobStore.deleteContainer(containerName);
88           context.close();
89        }
90     }

92     @Test
93     public void testListAllWhenOnePage() throws Exception {
94        BlobStore blobStore = createMock(BlobStore.class);
95        ListContainerOptions options = ListContainerOptions.NONE;
96        StorageMetadata v1 = createMock(StorageMetadata.class);
97        PageSet<StorageMetadata> pageSet = new PageSetImpl<StorageMetadata>(Collections.singletonList(v1), null);
98        
99        EasyMock.<PageSet<? extends StorageMetadata>>expect(blobStore.list(containerName, options)).andReturn(pageSet).once();
100       EasyMock.replay(blobStore);
101       
102       Iterable<StorageMetadata> iterable = BlobStores.listAll(blobStore, containerName, options);
103       assertEquals(ImmutableList.copyOf(iterable), ImmutableList.of(v1));
